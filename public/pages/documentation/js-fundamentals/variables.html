Global Variables: let and var

Why you should avoid it:

Let
let globalVariable = "I am a global variable";
function myFunction() {
let globalVariable = "I am a local variable";
console.log(globalVariable); // Output: "I am a local variable"
myFunction();
console.log(globalVariable); // Output: "I am a global variable"
Var
var x = 10;
function myFunction() {
var x = 20;
console.log(x); }
myFunction(); // Output: 20
console.log(x); // Output: 10
